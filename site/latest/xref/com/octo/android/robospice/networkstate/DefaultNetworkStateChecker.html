9   /**
10   * Consider network to be available when one network connection is up, whatever
11   * it is. This class will also ensure that both
12   * <tt>android.permission.ACCESS_NETWORK_STATE</tt> and
13   * <tt>android.permission.ACCESS_NETWORK</tt> are granted.
14   * @author sni
15   */
16  public class DefaultNetworkStateChecker implements NetworkStateChecker {
17  
18      @Override
19      public boolean isNetworkAvailable(final Context context) {
20          final ConnectivityManager connectivityManager = (ConnectivityManager) context
21              .getSystemService(Context.CONNECTIVITY_SERVICE);
22          final NetworkInfo[] allNetworkInfos = connectivityManager
23              .getAllNetworkInfo();
24          for (final NetworkInfo networkInfo : allNetworkInfos) {
25              if (networkInfo.getState() == NetworkInfo.State.CONNECTED
26                  || networkInfo.getState() == NetworkInfo.State.CONNECTING) {
27                  return true;
28              }
29          }
30          return false;
31      }
32  
33      @Override
34      public void checkPermissions(final Context context) {
35          checkHasPermission(context, Manifest.permission.ACCESS_NETWORK_STATE);
36          checkHasPermission(context, Manifest.permission.INTERNET);
37      }
38  
39      private boolean checkHasPermission(final Context context,
40          final String permissionName) {
41          final boolean hasPermission = context.getPackageManager()
42              .checkPermission(permissionName, context.getPackageName()) == PackageManager.PERMISSION_GRANTED;
43          if (!hasPermission) {
44              throw new SecurityException(
45                  "Application doesn\'t declare <uses-permission android:name=\""
46                      + permissionName + "\" />");
47          }
48          return true;
49      }
