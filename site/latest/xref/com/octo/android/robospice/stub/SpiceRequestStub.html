9   /**
10   * A {@link SpiceRequest} that is stubbed to provide testable state. Typically,
11   * when testing parts of RoboSpice, we will create a request and send it to the
12   * engine. As the engine is asynchronous the request is processed in a different
13   * thread than the testing thread. This stub allows to wait for request to be
14   * completed to make assertions in the test. It will also provide testable
15   * states (knowing if methods have been called or not).
16   * @author sni
17   * @param <T>
18   *            the type of the request's result.
19   */
20  public abstract class SpiceRequestStub<T> extends SpiceRequest<T> {
21      /** Whether {@link #loadDataFromNetwork()} has been called or not. */
22      protected boolean isLoadDataFromNetworkCalled = false;
23      /** Synchronizes access to {@link #requestFinishedCondition}. */
24      private ReentrantLock lock = new ReentrantLock();
25      /** Indicates whether or not request can be considered as finished. */
26      private Condition requestFinishedCondition = lock.newCondition();
27  
28      public SpiceRequestStub(Class<T> clazz) {
29          super(clazz);
30          setRetryPolicy(null);
31      }
32  
33      protected void signalStopWaiting() {
34          lock.lock();
35          try {
36              requestFinishedCondition.signal();
37          } finally {
38              lock.unlock();
39          }
40      }
41  
42      /**
43       * @return a boolean to indicate whether or not the
44       *         {@link #loadDataFromNetwork()} method has been called.
45       */
46      public boolean isLoadDataFromNetworkCalled() {
47          signalStopWaiting();
