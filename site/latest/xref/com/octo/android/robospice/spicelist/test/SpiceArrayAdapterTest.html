164         protected void loadBitmapAsynchronously(DataUnderTest octo, ImageView thumbImageView, String tempThumbnailImageFileName) {
165             super.loadBitmapAsynchronously(octo, thumbImageView, tempThumbnailImageFileName);
166             reentrantLock.lock();
167             try {
168                 loadBitmapHasBeenCalled = true;
169                 loadBitmapHasBeenCalledCondition.signal();
170             } finally {
171                 reentrantLock.unlock();
172             }
173         }

175         public void await(long millisecond) throws InterruptedException {
176             reentrantLock.lock();
177             try {
178                 loadBitmapHasBeenCalledCondition.await(millisecond, TimeUnit.MILLISECONDS);
179             } finally {
180                 reentrantLock.unlock();
181             }
182         }
