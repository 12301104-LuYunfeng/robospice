17  /**
18   * Stores / retrieves data in file system. This {@link ObjectPersister} is
19   * optimized for memory. It will only use file system to save the data, without
20   * allocating large memory space to transfert a given binary to the cache. It
21   * should be preferred to {@link InFileInputStreamObjectPersister} for low
22   * memory devices or memory intensive usages.
23   * @author SNI
24   */
25  public final class InFileBigInputStreamObjectPersister extends InFileInputStreamObjectPersister {

28          super(application);

33      }

35      @Override
36      public InputStream saveDataToCacheAndReturnData(InputStream data, Object cacheKey) throws CacheSavingException {
37          FileOutputStream output = null;
38          // special case for big inputstream object : as it can be read
39          // only once and is too big to be locally
40          // duplicated,
41          // 1) we save it in file
42          // 2) we load and return it from the file
43          try {
44              output = new FileOutputStream(getCacheFile(cacheKey));
45              IOUtils.copy(data, output);
46              return new FileInputStream(getCacheFile(cacheKey));
47          } catch (IOException e) {
48              throw new CacheSavingException(e);
49          } finally {
50              IOUtils.closeQuietly(output);
51          }
52      }
53  
54      @Override
55      public void setAsyncSaveEnabled(boolean isAsyncSaveEnabled) {
56          if (isAsyncSaveEnabled) {
57              throw new IllegalStateException("Asynchronous saving operation not supported.");
58          }
59      }
