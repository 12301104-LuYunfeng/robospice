18  /**
19   * An entity responsible for loading/saving data from/to cache. It implements a
20   * Chain of Responsibility pattern, delegating loading and saving operations to
21   * {@link ObjectPersister} or {@link ObjectPersisterFactory} elements. The chain
22   * of responsibility is ordered. This means that the order used to register
23   * elements matters. All elements in the chain of responsibility are questioned
24   * in order. The first element that can handle a given class for persistence
25   * will be used to persist data of this class.
26   * @author sni
27   */
28  /*
29   * Note to maintainers : concurrency must be taken care of as persister can be
30   * created by factories at any time. Thx to Henri Tremblay from EasyMock for
31   * peer review and concurrency checks.
32   */
33  @SuppressWarnings("deprecation")
34  public class CacheManager implements ICacheManager {
35  
36      /** The Chain of Responsibility list of all {@link Persister}. */
37      private Collection<Persister> listPersister = new ArrayList<Persister>();
38      private Map<ObjectPersisterFactory, List<ObjectPersister<?>>> mapFactoryToPersister = new HashMap<ObjectPersisterFactory, List<ObjectPersister<?>>>();
39  
40      /** {@inheritDoc} */
41      @Override
42      public void addPersister(Persister persister) {
43          listPersister.add(persister);
44          if (persister instanceof ObjectPersisterFactory) {
45              // will lead the list to be copied whenever we add a persister to it
46              // but there won't be any overhead while iterating through the list.
47              mapFactoryToPersister.put((ObjectPersisterFactory) persister, new CopyOnWriteArrayList<ObjectPersister<?>>());
48          } else if (!(persister instanceof ObjectPersister)) {
49              throw new RuntimeException(getClass().getSimpleName() + " only supports " + ObjectPersister.class.getSimpleName() + " or " + ObjectPersisterFactory.class.getSimpleName() + " instances.");
50          }
51      }
52  
53      /** {@inheritDoc} */
54      @Override
55      public void removePersister(Persister persister) {
56          listPersister.remove(persister);
57          if (persister instanceof ObjectPersisterFactory) {
58              mapFactoryToPersister.remove(persister);
59          }
60      }

146     /** {@inheritDoc} */
147     @Override
148     public void removeAllDataFromCache() {
149         for (Persister persister : this.listPersister) {
150             if (persister instanceof CacheCleaner) {
151                 ((CacheCleaner) persister).removeAllDataFromCache();
152             }
153 
154             if (persister instanceof ObjectPersisterFactory) {
155                 ObjectPersisterFactory factory = (ObjectPersisterFactory) persister;
156                 List<ObjectPersister<?>> listPersisterForFactory = mapFactoryToPersister.get(factory);
157                 for (ObjectPersister<?> objectPersister : listPersisterForFactory) {
158                     objectPersister.removeAllDataFromCache();
159                 }
160             }
161 
162         }
163     }

165     @SuppressWarnings("unchecked")
166     protected <T> ObjectPersister<T> getObjectPersister(Class<T> clazz) throws CacheCreationException {
167         for (Persister persister : this.listPersister) {
168             if (persister.canHandleClass(clazz)) {
169                 if (persister instanceof ObjectPersister) {
170                     return (ObjectPersister<T>) persister;
171                 }
172 
173                 if (persister instanceof ObjectPersisterFactory) {
174                     ObjectPersisterFactory factory = (ObjectPersisterFactory) persister;
175 
176                     if (factory.canHandleClass(clazz)) {
177                         List<ObjectPersister<?>> listPersisterForFactory = mapFactoryToPersister.get(factory);
178                         for (ObjectPersister<?> objectPersister : listPersisterForFactory) {
179                             if (objectPersister.canHandleClass(clazz)) {
180                                 return (ObjectPersister<T>) objectPersister;
181                             }
182                         }
183                         ObjectPersister<T> newPersister = factory.createObjectPersister(clazz);
184                         newPersister.setAsyncSaveEnabled(factory.isAsyncSaveEnabled());
185                         listPersisterForFactory.add(newPersister);
186                         return newPersister;
187                     }
188                 }
189             }
190         }
191         throw new RuntimeException("Class " + clazz.getName() + " is not handled by any registered factoryList");
192     }
