28      @Override
29      public String loadDataFromCache(Object cacheKey,
30          long maxTimeInCacheBeforeExpiry) throws CacheLoadingException {
31          Ln.v("Loading String for cacheKey = " + cacheKey);
32          File file = getCacheFile(cacheKey);
33          if (file.exists()) {
34              long timeInCache = System.currentTimeMillis() - file.lastModified();
35              if (maxTimeInCacheBeforeExpiry == 0
36                  || timeInCache <= maxTimeInCacheBeforeExpiry) {
37                  try {
38                      return FileUtils.readFileToString(file, CharEncoding.UTF_8);
39                  } catch (FileNotFoundException e) {
40                      // Should not occur (we test before if
41                      // file exists)
42                      // Do not throw, file is not cached
43                      Ln.w("file " + file.getAbsolutePath() + " does not exists",
44                          e);
45                      return null;
46                  } catch (Exception e) {
47                      throw new CacheLoadingException(e);
48                  }
49              }
50          }
51          Ln.v("file " + file.getAbsolutePath() + " does not exists");
52          return null;
53      }

55      @Override
56      public String saveDataToCacheAndReturnData(final String data,
57          final Object cacheKey) throws CacheSavingException {
58          Ln.v("Saving String " + data + " into cacheKey = " + cacheKey);
59          try {
60              if (isAsyncSaveEnabled()) {
61  
62                  Thread t = new Thread() {
63                      @Override
64                      public void run() {
65                          try {
66                              FileUtils.writeStringToFile(getCacheFile(cacheKey),
67                                  data, CharEncoding.UTF_8);
68                          } catch (IOException e) {
69                              Ln.e(e, "An error occured on saving request "
70                                  + cacheKey + " data asynchronously");
71                          }
72                      };
73                  };
74                  t.start();
75              } else {
76                  FileUtils.writeStringToFile(getCacheFile(cacheKey), data,
77                      CharEncoding.UTF_8);
78              }
79          } catch (Exception e) {
80              throw new CacheSavingException(e);
81          }
82          return data;
83      }
