48      @Override
49      public T loadDataFromCache(Object cacheKey, long maxTimeInCacheBeforeExpiry)
50          throws CacheLoadingException {
51          T result = null;
52  
53          File file = getCacheFile(cacheKey);
54          if (file.exists()) {
55              long timeInCache = System.currentTimeMillis() - file.lastModified();
56              if (maxTimeInCacheBeforeExpiry == 0
57                  || timeInCache <= maxTimeInCacheBeforeExpiry) {
58                  try {
59                      JsonParser jsonParser = jsonFactory
60                          .createJsonParser(new FileReader(getCacheFile(cacheKey)));
61                      result = jsonParser.parse(getHandledClass(), null);
62                      jsonParser.close();
63                      return result;
64                  } catch (FileNotFoundException e) {
65                      // Should not occur (we test before if file exists)
66                      // Do not throw, file is not cached
67                      Ln.w("file " + file.getAbsolutePath() + " does not exists",
68                          e);
69                      return null;
70                  } catch (Exception e) {
71                      throw new CacheLoadingException(e);
72                  }
73              }
74              Ln.v("Cache content is expired since "
75                  + (maxTimeInCacheBeforeExpiry - timeInCache));
76              return null;
77          }
78          Ln.v("file " + file.getAbsolutePath() + " does not exists");
79          return null;
80      }

82      @Override
83      public T saveDataToCacheAndReturnData(final T data, final Object cacheKey)
84          throws CacheSavingException {
85  
86          try {
87              if (isAsyncSaveEnabled()) {
88                  Thread t = new Thread() {
89                      @Override
90                      public void run() {
91                          try {
92                              saveData(data, cacheKey);
93                          } catch (IOException e) {
94                              Ln.e(e, "An error occured on saving request "
95                                  + cacheKey + " data asynchronously");
96                          } catch (CacheSavingException e) {
97                              Ln.e(e, "An error occured on saving request "
98                                  + cacheKey + " data asynchronously");
99                          }
100                     };
101                 };
102                 t.start();
103             } else {
104                 saveData(data, cacheKey);
105             }
106         } catch (CacheSavingException e) {
107             throw e;
108         } catch (Exception e) {
109             throw new CacheSavingException(e);
110         }
111         return data;
112     }

114     private void saveData(T data, Object cacheKey) throws IOException,
115         CacheSavingException {
116         // transform the content in json to store it in the cache
117         JsonGenerator jsonGenerator = jsonFactory
118             .createJsonGenerator(new FileWriter(getCacheFile(cacheKey)));
119         jsonGenerator.serialize(data);
120         jsonGenerator.flush();
121         jsonGenerator.close();
122     }
