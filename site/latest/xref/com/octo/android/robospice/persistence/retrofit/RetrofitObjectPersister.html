49      @Override
50      public T saveDataToCacheAndReturnData(final T data, final Object cacheKey) throws CacheSavingException {
51  
52          try {
53              if (isAsyncSaveEnabled()) {
54                  Thread t = new Thread() {
55                      @Override
56                      public void run() {
57                          try {
58                              saveData(data, cacheKey);
59                          } catch (IOException e) {
60                              Ln.e(e, "An error occured on saving request " + cacheKey + " data asynchronously");
61                          } catch (CacheSavingException e) {
62                              Ln.e(e, "An error occured on saving request " + cacheKey + " data asynchronously");
63                          }
64                      };
65                  };
66                  t.start();
67              } else {
68                  saveData(data, cacheKey);
69              }
70          } catch (CacheSavingException e) {
71              throw e;
72          } catch (Exception e) {
73              throw new CacheSavingException(e);
74          }
75          return data;
76      }

78      private void saveData(T data, Object cacheKey) throws IOException, CacheSavingException {
79          // transform the content in json to store it in the cache
80          TypedOutput typedBytes = converter.toBody(data);
81          FileOutputStream out = null;
82          try {
83              out = new FileOutputStream(getCacheFile(cacheKey));
84              typedBytes.writeTo(out);
85          } finally {
86              if (out != null) {
87                  out.close();
88              }
89          }
90      }

108 
109                 @Override
110                 public InputStream in() throws IOException {
111                     return new ByteArrayInputStream(body);
112                 }
113             };
114             return (T) converter.fromBody(typedInput, getHandledClass());
115         } catch (FileNotFoundException e) {
116             // Should not occur (we test before if file exists)
117             // Do not throw, file is not cached
118             Ln.w("file " + file.getAbsolutePath() + " does not exists", e);
