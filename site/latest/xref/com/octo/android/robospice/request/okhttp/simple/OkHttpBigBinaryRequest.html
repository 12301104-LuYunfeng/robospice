16  /**
17   * Downloads big images in size. All data is passed to the listener using file
18   * system. This class is meant to help download big images. If you wish to
19   * download smaller documents, you would be better using
20   * {@link OkHttpSmallBinaryRequest}.
21   * @author sni 
22   */
23  public class OkHttpBigBinaryRequest extends OkHttpBinaryRequest {
24  
25      protected File cacheFile;
26  
27      /**
28       * Creates a OkHttpBigBinaryRequest using its own cache file to prevent any downloaded data to be stored in memory.
29       * All file received from the network (via a simple http GET request) will be stored directly in the cache file to prevent memory loss.
30       * @param url the url to get the image data from.
31       * @param cacheFile a file used to store image data. Developers will have to handle the cache file erasure by themselves.
32       * <b>This cache file is not handled by RS caching mechanism in the "normal way". </b>
33       */
34      public OkHttpBigBinaryRequest(final String url, final File cacheFile) {
35          super(url);
36          this.cacheFile = cacheFile;
37      }
38  
39      @Override
40      public InputStream processStream(final int contentLength,
41          final InputStream inputStream) throws IOException {
42          OutputStream fileOutputStream = null;
43          try {
44              // touch
45              boolean isTouchedNow = cacheFile.setLastModified(System
46                  .currentTimeMillis());
47              if (!isTouchedNow) {
48                  Ln.d(
49                      "Modification time of file %s could not be changed normally ",
50                      cacheFile.getAbsolutePath());
51              }
52              fileOutputStream = new FileOutputStream(cacheFile);
53              readBytes(inputStream, new ProgressByteProcessor(this,
54                  fileOutputStream, contentLength));
55              return new FileInputStream(cacheFile);
56          } finally {
57              IOUtils.closeQuietly(fileOutputStream);
58          }
59      }
60  
61      public File getCacheFile() {
62          return cacheFile;
63      }
