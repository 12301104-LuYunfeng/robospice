62      /**
63       * Inspired from Guava com.google.common.io.ByteStreams
64       */
65      protected void readBytes(final InputStream in,
66          final ProgressByteProcessor processor) throws IOException {
67          final byte[] buf = new byte[BUF_SIZE];
68          try {
69              int amt;
70              do {
71                  amt = in.read(buf);
72                  if (amt == -1) {
73                      break;
74                  }
75              } while (processor.processBytes(buf, 0, amt));
76          } finally {
77              IOUtils.closeQuietly(in);
78          }
79      }
