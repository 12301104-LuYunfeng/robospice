80      private void checkInnerClassDeclarationToPreventMemoryLeak() {
81          // thanx to Cyril Mottier for this contribution
82          // prevent devs from creating memory leaks by using inner
83          // classes of contexts
84          if (getClass().isMemberClass() && Context.class.isAssignableFrom(getClass().getDeclaringClass()) && !Modifier.isStatic(getClass().getModifiers())) {
85              throw new IllegalArgumentException("Requests must be either non-inner classes or a static inner member class of Context : " + getClass());
