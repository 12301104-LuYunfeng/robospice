22  public abstract class OkHttpBinaryRequest extends OkHttpSpiceRequest<InputStream> {
23  
24      private static final int BUF_SIZE = 4096;
25      protected String url;
26  
27      public OkHttpBinaryRequest(final String url) {
28          super(InputStream.class);
29          this.url = url;
30      }
31  
32      @Override
33      public final InputStream loadDataFromNetwork() throws Exception {
34          try {
35              OkUrlFactory urlFactory = new OkUrlFactory(getOkHttpClient());
36              HttpURLConnection connection = urlFactory.open(new URL(url));
37              return processStream(connection.getContentLength(), connection.getInputStream());
38          } catch (final MalformedURLException e) {
39              Ln.e(e, "Unable to create URL");
40              throw e;
41          } catch (final IOException e) {
42              Ln.e(e, "Unable to download binary");
43              throw e;
44          }
45      }
46  
47      protected final String getUrl() {
48          return this.url;
49      }
50  
51      /**
52       * Override this method to process the stream downloaded
53       * @param contentLength
54       *            size of the download
55       * @param inputStream
56       *            stream of the download
57       * @return an inputstream containing the download
58       */
59      public abstract InputStream processStream(int contentLength, InputStream inputStream) throws IOException;
60  
61      /**
62       * Inspired from Guava com.google.common.io.ByteStreams
63       */
64      protected void readBytes(final InputStream in, final ProgressByteProcessor processor) throws IOException {
65          final byte[] buf = new byte[BUF_SIZE];
66          try {
67              int amt;
68              do {
69                  amt = in.read(buf);
70                  if (amt == -1) {
71                      break;
72                  }
73              } while (processor.processBytes(buf, 0, amt));
74          } finally {
75              IOUtils.closeQuietly(in);
76          }
77      }
