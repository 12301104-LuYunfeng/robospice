6   /**
7    * A ThreadPoolExecutor with priority management, from <a href=
8    * "http://stackoverflow.com/q/807223/693752">SOF</a> <br/>
9    * There are some inherent limitations to this mechanism. For instance, the
10   * first runnable/callable passed to the executor doesn't go in the queue. Thus
11   * the priority mechanism will only apply when tasks are queued, and this
12   * happens when the number of current runners is exceeds the number of max
13   * thread in the pool size.
14   * @author SNI
15   */
16  public final class PriorityThreadPoolExecutor extends PausableThreadPoolExecutor {

24      }

27          super(poolSize, threadPriority);
28      }

31          super(poolSize);

35      // API

44          return new PriorityThreadPoolExecutor(nThreads, threadPriority);

51      // ----------------------------------
52      // OVVERRIDEN METHODS
53      // ----------------------------------
54      @Override
55      // thanks to
56      // http://binkley.blogspot.fr/2009/04/jumping-work-queue-in-executor.html
57      // for backward compat to SDK 8 (JDK 5)
58      protected <T> RunnableFuture<T> newTaskFor(Runnable runnable, T value) {
59          // makes findbugs happy, there must be some @Nullable
60          // annotation in JDK..
61          if (runnable == null) {
62              return null;
63          }
64          return new PriorityFuture<T>(runnable, ((PriorityRunnable) runnable).getPriority(), value);
65      }
66  
67      // form JDK 1.6, to ensure backward compatibility
68      @Override
69      public Future<?> submit(Runnable task) {
70          if (task == null) {
71              throw new NullPointerException();
72          }
73          RunnableFuture<Object> ftask = newTaskFor(task, null);
74          execute(ftask);
75          return ftask;
76      }
