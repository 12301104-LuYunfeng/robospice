93              cacheManager, executorService, requestProcessorListener,

95          requestProcessor.setFailOnCacheError(DEFAULT_FAIL_ON_CACHE_ERROR);

126     /**
127      * Factory method to create an {@link ExecutorService} that will be used to
128      * execute requests. The default implementation of this method will create a
129      * single threaded or multi-threaded {@link ExecutorService} depending on
130      * the number of threads returned by {@link #getThreadCount()}. If you
131      * override this method in your service, you can supply a custom
132      * {@link ExecutorService}. This feature has been implemented following a
133      * request from Riccardo Ciovati.
134      * @return the {@link ExecutorService} to be used to execute requests .
135      */
136     protected ExecutorService getExecutorService() {
137         ExecutorService executorService;
138         final int threadCount = getThreadCount();
139         if (threadCount <= 0) {
140             throw new IllegalArgumentException("Thread count must be >= 1");
141         } else if (threadCount == 1) {
142             executorService = Executors.newSingleThreadExecutor();
143         } else {
144             executorService = Executors.newFixedThreadPool(threadCount,
145                 new ThreadFactory() {
146 
147                     @Override
148                     public Thread newThread(final Runnable r) {
149                         return new Thread(r);
150                     }
151                 });
152         }
153         return executorService;
154     }

290     // There is a bug in ServiceTestCase : a call to setForeground will fail
291     // http://code.google.com/p/android/issues/detail?id=12122
292     private void startForeground(final Notification notification) {
293         try {
294             final Method setForegroundMethod = Service.class.getMethod(
295                 "startForeground", int.class, Notification.class);
296             setForegroundMethod.invoke(this, NOTIFICATION_ID, notification);
297         } catch (final SecurityException e) {
298             Ln.e(e, "Unable to start a service in foreground");
299         } catch (final NoSuchMethodException e) {
300             Ln.e(e, "Unable to start a service in foreground");
301         } catch (final IllegalArgumentException e) {
302             Ln.e(e, "Unable to start a service in foreground");
303         } catch (final IllegalAccessException e) {
304             Ln.e(e, "Unable to start a service in foreground");
305         } catch (final InvocationTargetException e) {
306             Ln.e(e, "Unable to start a service in foreground");
307         }
